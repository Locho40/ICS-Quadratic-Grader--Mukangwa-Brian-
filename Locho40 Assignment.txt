<script>
    function solve() {
        document.getElementById('errA').textContent = '';
        document.getElementById('errB').textContent = '';
        document.getElementById('errC').textContent = '';
        document.getElementById('output').textContent = '';

        let a = parseFloat(document.getElementById('a').value);
        let b = parseFloat(document.getElementById('b').value);
        let c = parseFloat(document.getElementById('c').value);
        let valid = true;

        if (isNaN(a)) {
            document.getElementById('errA').textContent = "Enter a number for a.";
            valid = false;
        } else if (a === 0) {
            document.getElementById('errA').textContent = "a cannot be 0.";
            valid = false;
        }
        if (isNaN(b)) {
            document.getElementById('errB').textContent = "Enter a number for b.";
            valid = false;
        }
        if (isNaN(c)) {
            document.getElementById('errC').textContent = "Enter a number for c.";
            valid = false;
        }
        if (!valid) return;

        let D = b*b - 4*a*c;
        let res = "Discriminant (D) = " + D.toFixed(2) + "\n";

        if (D > 0) {
            let r1 = (-b + Math.sqrt(D)) / (2*a);
            let r2 = (-b - Math.sqrt(D)) / (2*a);
            res += "Two distinct roots: x1 = " + r1.toFixed(2) + ", x2 = " + r2.toFixed(2);
        } else if (D === 0) {
            let r = -b/(2*a);
            res += "One repeated root: x = " + r.toFixed(2);
        } else {
            let real = (-b/(2*a)).toFixed(2);
            let imag = (Math.sqrt(-D)/(2*a)).toFixed(2);
            res += "Complex roots: x1 = " + real + " + " + imag + "i, x2 = " + real + " - " + imag + "i";
        }

        document.getElementById('output').textContent = res;
    }

    function clearFields() {
        document.getElementById('a').value = '';
        document.getElementById('b').value = '';
        document.getElementById('c').value = '';
        document.getElementById('output').textContent = '';
        document.getElementById('errA').textContent = '';
        document.getElementById('errB').textContent = '';
        document.getElementById('errC').textContent = '';
    }

    function getGrade() {
        document.getElementById('scoreErr').textContent = '';
        document.getElementById('gradeOutput').textContent = '';

        let score = parseInt(document.getElementById('score').value);
        if (isNaN(score) || score < 0 || score > 100) {
            document.getElementById('scoreErr').textContent = "Enter a number between 0 and 100.";
            return;
        }

        let grade = "";
        if (score >= 85) grade = "A+";
        else if (score >= 75) grade = "A";
        else if (score >= 65) grade = "B+";
        else if (score >= 60) grade = "B";
        else if (score >= 55) grade = "C+";
        else if (score >= 50) grade = "C";
        else grade = "D";

        document.getElementById('gradeOutput').textContent = "Score " + score + " - Grade " + grade;
    }

    function clearScore() {
        document.getElementById('score').value = '';
        document.getElementById('scoreErr').textContent = '';
        document.getElementById('gradeOutput').textContent = '';
    }
