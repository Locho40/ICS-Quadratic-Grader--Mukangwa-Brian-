    <script>
        function solveQuadratic() {
            const a = parseFloat(document.getElementById('a').value);
            const b = parseFloat(document.getElementById('b').value);
            const c = parseFloat(document.getElementById('c').value);
            let result = "";

            if (a === 0) {
                if (b === 0) {
                    result = c === 0 ? "Infinite solutions" : "No solution";
                } else {
                    const x = -c / b;
                    result = "Linear solution: x = " + x;
                }
            } else {
                const discriminant = b*b - 4*a*c;
                if (discriminant > 0) {
                    const x1 = (-b + Math.sqrt(discriminant)) / (2*a);
                    const x2 = (-b - Math.sqrt(discriminant)) / (2*a);
                    result = Two real solutions: x₁ = ${x1}, x₂ = ${x2};
                } else if (discriminant === 0) {
                    const x = -b / (2*a);
                    result = One real solution: x = ${x};
                } else {
                    const realPart = (-b / (2*a)).toFixed(2);
                    const imagPart = (Math.sqrt(-discriminant) / (2*a)).toFixed(2);
                    result = Two complex solutions: x₁ = ${realPart} + ${imagPart}i, x₂ = ${realPart} - ${imagPart}i;
                }
            }

            document.getElementById('quadraticResult').textContent = result;
        }

        function convertGrade() {
            const score = parseFloat(document.getElementById('score').value);
            let grade = "";

            if (score >= 85 && score <= 100) grade = "A+";
            else if (score >= 75 && score <= 84) grade = "A";
            else if (score >= 65 && score <= 74) grade = "B+";
            else if (score >= 60 && score <= 64) grade = "B";
            else if (score >= 55 && score <= 59) grade = "C+";
            else if (score >= 50 && score <= 54) grade = "C";
            else if (score >= 0 && score <= 49) grade = "D";
            else grade = "Invalid score";

            document.getElementById('gradeResult').textContent = "Letter Grade: " + grade;
        }
    </script>
